outliers_fraction = 0.0183
xx, yy = np.meshgrid(np.linspace(0, 1, 100), np.linspace(0, 1, 100))

# copy of dataframe
df1 = df
nb_id = df1['CarId'].nunique()+1
df1['outlier'] = df1['Class']

# CarId - inlier feature 1,  Speed_diff - inlier feature 2
inliers_CarId = np.array(df1['CarId'][df1['outlier'] == 0]).reshape(-1,1)
inliers_Speed_diff = np.array(df1['Speed diff'][df1['outlier'] == 0]).reshape(-1,1)
    
# CarId - outlier feature 1, Speed_diff - outlier feature 2
outliers_CarId = df1['CarId'][df1['outlier'] == 1].values.reshape(-1,1)
outliers_Speed_diff = df1['Speed diff'][df1['outlier'] == 1].values.reshape(-1,1)

classifiers = {
    'Cluster-based Local Outlier Factor (CBLOF)':CBLOF(contamination=outliers_fraction, check_estimator=False, random_state=0, n_clusters=nb_id),
    'Isolation Forest': IForest(contamination=outliers_fraction, random_state=0),
    'Locally Selective Combination (LSCP)': LSCP(detector_list=[MCD(),MCD()], contamination=outliers_fraction, random_state=0),
}

X = X_speed
plt.figure(figsize=(30, 30))
for i, (clf_name, clf) in enumerate(classifiers.items()):
    clf.fit(X)
    scores_pred = clf.decision_function(X) * -1
    y_pred = clf.predict(X)
    threshold = percentile(scores_pred, 100 * outliers_fraction)
        
    # fill blue map colormap from minimum anomaly score to threshold value
    Z = clf.decision_function(np.c_[xx.ravel(), yy.ravel()]) * -1
    Z = Z.reshape(xx.shape)
    subplot = plt.subplot(3, 4, i + 1)
    subplot.contourf(xx, yy, Z, levels=np.linspace(Z.min(), threshold, 7), cmap=plt.cm.Blues_r)
        
    # draw red contour line where anomaly score is equal to thresold
    a = subplot.contour(xx, yy, Z, levels=[threshold], linewidths=2, colors='red')
    subplot.contourf(xx, yy, Z, levels=[threshold, Z.max()], colors='orange')
        
    # fill orange contour lines where range of anomaly score is from threshold to maximum anomaly score
    b = subplot.scatter(inliers_CarId, inliers_Speed_diff, c='white', s=20, edgecolor='k')
    c = subplot.scatter(outliers_CarId, outliers_Speed_diff, c='black', s=20, edgecolor='k')
        
    subplot.axis('tight')
    subplot.legend(
        [a.collections[0], b, c],
        ['learned decision function', 'true inliers', 'true outliers'],
        prop=matplotlib.font_manager.FontProperties(size=10),
        loc='upper right')
    subplot.set_xlabel("%d. %s" % (i + 1, clf_name))
    subplot.set_xlim((0, 1))
    subplot.set_ylim((0, 1))
plt.show()

nb_id = df['CarId'].nunique()
df1 = df
df1['outlier'] = df1['Class']

X_all = pd.DataFrame(df1, columns=['ID', 'CarId', 'Speed diff', 'Heading diff', 'Position diff'])
X_all = X_all.to_numpy()
y_all = df1['Class'].to_numpy()

XS=np.size(X_all[:,0])
Y1=(y_all[:] == 1).sum()
Y2=(y_all[:] == 0).sum()

print("décompte des données :", XS)
print("donnée anormale :", Y1)
print("donnée normale :", Y2)

X_all, y_all = shuffle(X_all, y_all)  # Shuffle data.
iterator = ArrayStreamer(shuffle=False)  # Create streamer to simulate streaming data.
    
auroc = AUROCMetric()
aupr = AUPRMetric()

models = [ReferenceWindowModel(model_cls=CBLOF, window_size=2000, sliding_size=40, initial_window_X=X_all[:2000][:,[1,2]],n_clusters=nb_id),
         ReferenceWindowModel(model_cls=CBLOF, window_size=2000, sliding_size=40, initial_window_X=X_all[:2000][:,[1,3]],n_clusters=nb_id),
         ReferenceWindowModel(model_cls=CBLOF, window_size=2000, sliding_size=40, initial_window_X=X_all[:2000][:,[1,4]],n_clusters=nb_id)]

ensembler = MedianScoreEnsembler()  # Ensembler module.
recup = np.empty((0,3))  # Will get the result of the algorithm

for X, y in tqdm(iterator.iter(X_all, y_all)):  # Iterate over examples.
    model_scores = np.empty(len(models), dtype=np.float)
    # Fit & Score via for each model.
    for i, model in enumerate(models):
        if i == 0 :
            model.fit_partial(X[[1,2]])
            model_scores[i] = model.score_partial(X[[1,2]])
        if i == 1 :
            model.fit_partial(X[[1,3]])
            model_scores[i] = model.score_partial(X[[1,2]])
        if i == 2 :
            model.fit_partial(X[[1,4]])
            model_scores[i] = model.score_partial(X[[1,2]])
            
    score = ensembler.fit_transform_partial(model_scores)  # fit to ensembler model and get ensembled score.
    recup = np.append(recup, np.array([[X[0],y,score[0]]]), axis=0)

    auroc.update(y, score)  # update AUROC metric.
    aupr.update(y, score)  # update AUPR metric.

a = np.argsort(recup[:,-1])
recup = recup[a]
recup = recup[::-1]
np.savetxt("data/result/recup.csv", recup, delimiter=",", fmt='%f')

print("Window_size=2000")
print("AUROC: ", auroc.get())
print("AUPR: ", aupr.get())
